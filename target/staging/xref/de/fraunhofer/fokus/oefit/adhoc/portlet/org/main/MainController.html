108 	/**
109 	 * Adds the offer.
110 	 *
111 	 * @param data the data
112 	 * @param result the result
113 	 * @param request the request
114 	 * @param response the response
115 	 * @param model the model
116 	 */
117 	@ActionMapping(params = "action=addOffer")
118 	public void addOffer(@Valid @ModelAttribute("data") final OfferForm data,
119 	        final BindingResult result, final ActionRequest request,
120 	        final ActionResponse response, final Model model) {
121 		m_objLog.debug("addOffer::start");
122 
123 		final ThemeDisplay themeDisplay = this.getThemeDisplay(request);
124 
125 		this.m_objOfferFormValidator.setThemeDisplay(themeDisplay);
126 		this.m_objOfferFormValidator.validate(data, result);
127 
128 		if (!result.hasErrors()) {
129 			final long orgId = CustomOrgServiceHandler
130 			        .getOrgIdByLiferayUser(themeDisplay);
131 			data.setOrgId(orgId);
132 			CustomOfferServiceHandler.addOffer(data);
133 			data.clear();
134 		} else {
135 			model.addAttribute("countries", CustomPersistanceServiceHandler
136 			        .getDataList(E_CategoryType.COUNTRIES, true));
137 			model.addAttribute("workhours", CustomPersistanceServiceHandler
138 			        .getDataList(E_CategoryType.OFFERTIME, false));
139 			response.setRenderParameter("jspPage", "../shared/offer");
140 		}
141 
142 		m_objLog.debug("addOffer::end");
143 	}

146 	 * Adds the user.
147 	 *
148 	 * @param data the data
149 	 * @param result the result
150 	 * @param request the request
151 	 * @param response the response
152 	 * @param model the model
153 	 */
154 	@ActionMapping(params = "action=addUser")
155 	public void addUser(
156 	        @Valid @ModelAttribute("userData") final ProfileForm data,
157 	        final BindingResult result, final ActionRequest request,
158 	        final ActionResponse response, final Model model) {
159 		final ThemeDisplay themeDisplay = (ThemeDisplay) request
160 		        .getAttribute(WebKeys.THEME_DISPLAY);
161 		final long orgId = CustomOrgServiceHandler
162 		        .getOrgIdByLiferayUser(themeDisplay);
163 		m_objLog.debug("addUser::start(" + orgId + ")");
164 
165 		this.m_objProfileFormValidator.setThemeDisplay(themeDisplay);
166 		this.m_objProfileFormValidator.validate(data, result);
167 
168 		if (!result.hasErrors() && orgId >= 0) {
169 			final AHOrg org = AHOrgLocalServiceUtil
170 			        .getOrganisationByOwnerMail(themeDisplay.getUser()
171 			                .getEmailAddress());
172 			if (org != null) {
173 				final User newUser = CustomPortalServiceHandler
174 				        .createPortalUser(data.getForename(),
175 				                data.getSurname(), data.getMail(),
176 				                themeDisplay.getCompanyId(),
177 				                themeDisplay.getLocale());
178 				if (newUser != null) {
179 					AHOrgLocalServiceUtil.addOrganisationUser(orgId,
180 					        data.getMail());
181 				}
182 			}
183 			data.clear();
184 		}
185 
186 		m_objLog.debug("addUser::end");
187 	}

219 	 * Delete offer.
220 	 *
221 	 * @param request the request
222 	 * @param response the response
223 	 * @param model the model
224 	 */
225 	@ActionMapping(params = "action=deleteOffer")
226 	public void deleteOffer(final ActionRequest request,
227 	        final ActionResponse response, final Model model) {
228 		
229 		if (Constants.RESTRICT_TO_DEMO) {
230 			m_objLog.debug("deleteOffer::denied()");
231 			SessionErrors.add(request, "common.demo.denied");
232 			return;
233 		}
234 			
235 		
236 		final Long offerId = this.getOfferId(request);
237 		m_objLog.debug("deleteOffer::start(" + offerId + ")");
238 
239 		final ThemeDisplay themeDisplay = (ThemeDisplay) request
240 		        .getAttribute(WebKeys.THEME_DISPLAY);
241 
242 		if (!CustomLockServiceHandler.isLocked(AHOffer.class.getName(),
243 		        offerId, themeDisplay)) {
244 			try {
245 				AHOfferLocalServiceUtil.deleteAHOffer(offerId);
246 				// force unlock
247 				CustomLockServiceHandler.unlock(AHOffer.class.getName(),
248 				        offerId, themeDisplay);
249 			} catch (final Throwable e) {
250 				m_objLog.error(e);
251 			}
252 		}
253 
254 		m_objLog.debug("deleteOffer::end");
255 	}

310 		}
311 
312 		m_objLog.debug("editOffer::end");

347 	 * Handle exception.
348 	 *
349 	 * @param ex the ex
350 	 * @param request the request
351 	 * @param response the response
352 	 * @return the model and view
353 	 */
354 	@ExceptionHandler(MaxUploadSizeExceededException.class)
355 	public ModelAndView handleException(final Exception ex,
356 	        final RenderRequest request, final RenderResponse response) {
357 		m_objLog.error(ex);
358 		final ModelAndView mv = new ModelAndView(DEFAULT_PAGE);
359 
360 		// response.setParameter("tabId", "profile");
361 		// response.setParameter("setModel", Boolean.FALSE.toString());
362 		final ThemeDisplay themeDisplay = (ThemeDisplay) request
363 		        .getAttribute(WebKeys.THEME_DISPLAY);
364 		final long orgId = CustomOrgServiceHandler
365 		        .getOrgIdByLiferayUser(themeDisplay);
366 		RegistrationForm formData = null;
367 		if (orgId >= 0) {
368 			formData = CustomOrgServiceHandler.getOrganisationForEdit(orgId);
369 		}
370 		if (formData != null) {
371 			mv.addObject("orgData", formData);
372 		}
373 		mv.addObject("countries", CustomPersistanceServiceHandler.getDataList(
374 		        E_CategoryType.COUNTRIES, true));
375 		mv.addObject("userData", new ProfileForm());
376 		mv.addObject("actionType", "edit");
377 		mv.addObject("tabId", "profile");
378 		SessionErrors.add(request, "org.form.addOrg.field.logoFile.exceed");
379 		return mv;
380 	}

392 	 * Prepare offer.
393 	 *
394 	 * @param request the request
395 	 * @param response the response
396 	 * @param model the model
397 	 */
398 	@ActionMapping(params = "action=prepareOffer")
399 	public void prepareOffer(final ActionRequest request,
400 	        final ActionResponse response, final Model model) {
401 		m_objLog.debug("prepareOffer::start()");
402 
403 		final ThemeDisplay themeDisplay = this.getThemeDisplay(request);
404 		final AHOrg org = CustomOrgServiceHandler
405 		        .getOrgByLiferayUser(themeDisplay);
406 		AHContact contact = null;
407 
408 		final OfferForm form = new OfferForm();
409 		if (org != null) {
410 			try {
411 				contact = AHContactLocalServiceUtil.getAHContact(org
412 				        .getContactId());
413 			} catch (final Throwable e) {
414 				m_objLog.warn(e);
415 			}
416 			if (contact != null) {
417 				form.setContactSndForename(contact.getForename());
418 				form.setContactSndSurname(contact.getSurname());
419 				form.setContactSndMail(contact.getEmail());
420 				form.setContactSndTel(contact.getTel());
421 			}
422 		}
423 		final String siteContactSurname = CustomPortalServiceHandler
424 		        .getConfigValue(E_ConfigKey.MGMT_CONTACT_SURNAME);
425 		final String siteContactForename = CustomPortalServiceHandler
426 		        .getConfigValue(E_ConfigKey.MGMT_CONTACT_FORENAME);
427 		final String siteContactMail = CustomPortalServiceHandler
428 		        .getConfigValue(E_ConfigKey.MGMT_CONTACT_MAIL);
429 		final String siteContactTel = CustomPortalServiceHandler
430 		        .getConfigValue(E_ConfigKey.MGMT_CONTACT_TEL);
431 		form.setContactForename(siteContactForename);
432 		form.setContactSurname(siteContactSurname);
433 		form.setContactMail(siteContactMail);
434 		form.setContactTel(siteContactTel);
435 
436 		model.addAttribute("data", form);
437 		model.addAttribute("countries", CustomPersistanceServiceHandler
438 		        .getDataList(E_CategoryType.COUNTRIES, true));
439 		model.addAttribute("workhours", CustomPersistanceServiceHandler
440 		        .getDataList(E_CategoryType.OFFERTIME, false));
441 
442 		response.setRenderParameter("jspPage", "../shared/offer");
443 
444 		m_objLog.debug("prepareOffer::end");
445 	}

448 	 * Render.
449 	 *
450 	 * @param request the request
451 	 * @param response the response
452 	 * @param model the model
453 	 * @return the string
454 	 */
455 	@RenderMapping
456 	public String render(final RenderRequest request,
457 	        final RenderResponse response, final Model model) {
458 		m_objLog.trace("render::start");
459 		String page = request.getParameter("jspPage");
460 		if (page == null) {
461 			page = DEFAULT_PAGE;
462 		}
463 		final String error = request.getParameter("error");
464 		if (error != null) {
465 			model.addAttribute("error", error);
466 			m_objLog.warn("Handing over error " + error);
467 			final String ePage = request.getParameter("errorPage");
468 			if (ePage != null) {
469 				page = ePage;
470 			}
471 		} /*
472 		 * else { model.addAttribute("data", new BerliosProject()); }
473 		 */
474 		final String sm = request.getParameter("setModel");
475 		boolean setModel = true;
476 		if (sm != null) {
477 			setModel = Boolean.parseBoolean(sm);
478 		}
479 		if (setModel) {
480 			final ThemeDisplay themeDisplay = (ThemeDisplay) request
481 			        .getAttribute(WebKeys.THEME_DISPLAY);
482 			final long orgId = CustomOrgServiceHandler
483 			        .getOrgIdByLiferayUser(themeDisplay);
484 			RegistrationForm formData = null;
485 			if (orgId >= 0) {
486 				formData = CustomOrgServiceHandler
487 				        .getOrganisationForEdit(orgId);
488 			}
489 			if (formData != null) {
490 				model.addAttribute("orgData", formData);
491 			}
492 		}
493 		model.addAttribute("actionType", "edit");
494 		m_objLog.trace("render::end(" + page + ")");
495 		return page;
496 	}

559

561 		response.setRenderParameter("setModel", Boolean.FALSE.toString());
562 		m_objLog.debug("saveOrganisation::end");
