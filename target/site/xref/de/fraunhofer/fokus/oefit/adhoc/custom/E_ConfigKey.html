283 	/**
284 	 * Gets the by category.
285 	 *
286 	 * @param cat the cat
287 	 * @return the by category
288 	 */
289 	public static List<E_ConfigKey> getByCategory(final E_ConfigCategory cat) {
290 		final List<E_ConfigKey> result = new LinkedList<E_ConfigKey>();
291 
292 		for (final E_ConfigKey key : E_ConfigKey.values()) {
293 			if (key.getCategory().equals(cat)) {
294 				result.add(key);
295 			}
296 		}
297 
298 		return result;
299 	}

308 	private E_ConfigKey(final E_ConfigCategory category,
309 	        final E_ConfigType type, final String msgProperty,

311 		this.m_strDefVal = defValue;

314 		this.m_objType = type;

318 	        final E_ConfigType type, final String msgProperty,
319 	        final String defValue, final String systemProperty) {
320 		this.m_strDefVal = defValue;
321 		this.m_strMsgProp = msgProperty;
