36  import java.util.List;
37  
38  import com.liferay.portal.kernel.log.Log;
39  import com.liferay.portal.kernel.log.LogFactoryUtil;
40  import com.liferay.portal.kernel.messaging.Message;
41  import com.liferay.portal.kernel.messaging.MessageListener;
42  import com.liferay.portal.kernel.messaging.MessageListenerException;
43  
44  import de.fraunhofer.fokus.oefit.adhoc.custom.CustomServiceUtils;
45  import de.fraunhofer.fokus.oefit.particity.model.AHOffer;
46  import de.fraunhofer.fokus.oefit.particity.model.AHOrg;
47  import de.fraunhofer.fokus.oefit.particity.model.custom.ModelNotificationRepository;
48  import de.fraunhofer.fokus.oefit.particity.service.AHOfferLocalServiceUtil;
49  import de.fraunhofer.fokus.oefit.particity.service.AHOrgLocalServiceUtil;
50  
51  /**
52   * Scheduler for hourly tasks
53   */
54  public class HourlyScheduler implements MessageListener {
55  
56  	private static final Log	m_objLog	= LogFactoryUtil
57  	                                             .getLog(HourlyScheduler.class);
58  
59  	private static final long	HOUR	 = 1000 * 60 * 60;
60  
61  	private void notifyExpiredOffers() {
62  		try {
63  			final long now = CustomServiceUtils.time();
64  			final long minExpired = now - HOUR;
65  			final int orgSize = AHOrgLocalServiceUtil.getAHOrgsCount();
66  			if (orgSize > 0) {
67  				List<AHOrg> orgs;
68  				List<AHOffer> offer;
69  				for (int i = 0; i < orgSize; i += 5) {
70  					orgs = AHOrgLocalServiceUtil.getAHOrgs(i, i + 5);
71  					for (final AHOrg org : orgs) {
72  						offer = AHOfferLocalServiceUtil
73  						        .findExpiredOffersForOrg(org.getOrgId(),
74  						                minExpired, now);
75  						if (offer != null && offer.size() > 0) {
76  							ModelNotificationRepository.getInstance()
77  							        .notifyOutdated(offer, HOUR);
78  						}
79  					}
80  				}
81  			}
82  		} catch (final Throwable t) {
83  			m_objLog.warn(
84  			        "Check for expired offers failed due to " + t.getMessage(),
85  			        t);
86  		}
87  	}
88  
89  	/* (non-Javadoc)
90  	 * @see com.liferay.portal.kernel.messaging.MessageListener#receive(com.liferay.portal.kernel.messaging.Message)
91  	 */
92  	@Override
93  	public void receive(final Message message) throws MessageListenerException {
94  		try {
95  
96  			this.notifyExpiredOffers();
97  
98  			m_objLog.info("Hourly scheduler update done.");
99  		} catch (final Throwable t) {
100 			m_objLog.warn(
101 			        "Could not exec daily schedule due to: " + t.getMessage(),
102 			        t);
103 		}
104 	}
105 
106 }
