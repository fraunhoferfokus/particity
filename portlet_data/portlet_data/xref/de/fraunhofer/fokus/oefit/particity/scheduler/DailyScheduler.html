85  	/**
86  	 * Removes the offers without addresses.
87  	 */
88  	public void removeOffersWithoutAddresses() {
89  		try {
90  			final int offerSize = AHOfferLocalServiceUtil.getAHOffersCount();
91  			if (offerSize > 0) {
92  				List<AHOffer> offers;
93  				AHAddr addr;
94  				for (int i = 0; i < offerSize; i += 5) {
95  					offers = AHOfferLocalServiceUtil.getAHOffers(i, i + 5);
96  					for (final AHOffer offer : offers) {
97  						try {
98  							addr = AHAddrLocalServiceUtil.getAHAddr(offer
99  							        .getAdressId());
100 						} catch (final Throwable t) {
101 							addr = null;
102 						}
103 						if (addr == null) {
104 							AHOfferLocalServiceUtil.deleteAHOffer(offer
105 							        .getOfferId());
106 							m_objLog.warn("Removing offer "
107 							        + offer.getOfferId() + "/"
108 							        + offer.getTitle()
109 							        + " due to missing address "
110 							        + offer.getAdressId());
111 						}
112 					}
113 				}
114 			}
115 		} catch (final Throwable t) {
116 			m_objLog.warn(
117 			        "Check for orphaned offers failed due to " + t.getMessage(),
118 			        t);
119 		}
120 	}

122 	/**
123 	 * Removes the offers without organisations.
124 	 */
125 	public void removeOffersWithoutOrganisations() {
126 		try {
127 			final int offerSize = AHOfferLocalServiceUtil.getAHOffersCount();
128 			if (offerSize > 0) {
129 				List<AHOffer> offers;
130 				AHOrg org;
131 				for (int i = 0; i < offerSize; i += 5) {
132 					offers = AHOfferLocalServiceUtil.getAHOffers(i, i + 5);
133 					for (final AHOffer offer : offers) {
134 						final long orgId = offer.getOrgId();
135 						try {
136 							org = AHOrgLocalServiceUtil.getAHOrg(orgId);
137 						} catch (final Throwable t) {
138 							org = null;
139 						}
140 						if (org == null) {
141 							AHOfferLocalServiceUtil.deleteAHOffer(offer
142 							        .getOfferId());
143 							m_objLog.warn("Removing offer "
144 							        + offer.getOfferId() + "/"
145 							        + offer.getTitle()
146 							        + " due to missing organisation "
147 							        + offer.getOrgId());
148 						}
149 					}
150 				}
151 			}
152 		} catch (final Throwable t) {
153 			m_objLog.warn(
154 			        "Check for orphaned offers failed due to " + t.getMessage(),
155 			        t);
156 		}
157 	}

159 	/**
160 	 * Removes the organisations without users.
161 	 */
162 	public void removeOrganisationsWithoutUsers() {
163 		try {
164 			final long companyId = PortalUtil.getDefaultCompanyId();
165 			final int orgSize = AHOrgLocalServiceUtil.getAHOrgsCount();
166 			if (orgSize > 0) {
167 				List<AHOrg> orgs;
168 				for (int i = 0; i < orgSize; i += 5) {
169 					orgs = AHOrgLocalServiceUtil.getAHOrgs(i, i + 5);
170 					for (final AHOrg org : orgs) {
171 						final String mail = org.getOwner();
172 						User user = null;
173 						try {
174 							user = UserLocalServiceUtil.getUserByEmailAddress(
175 							        companyId, mail);
176 						} catch (final Throwable t) {
177 						}
178 						if (user == null) {
179 							m_objLog.warn("Removing organisation "
180 							        + org.getOrgId() + "/" + org.getName()
181 							        + " due to missing user.");
182 							AHOrgLocalServiceUtil.deleteOrganisation(org
183 							        .getOrgId());
184 						}
185 					}
186 				}
187 			}
188 		} catch (final Throwable t) {
189 			m_objLog.warn(
190 			        "Check for inactive organisations failed due to "
191 			                + t.getMessage(), t);
192 		}
193 	}

195 	/**
196 	 * Removes the orphaned addresses.
197 	 */
198 	public void removeOrphanedAddresses() {
199 		try {
200 			final int addrSize = AHAddrLocalServiceUtil.getAHAddrsCount();
201 			if (addrSize > 0) {
202 				List<AHAddr> addresses;
203 				for (int i = 0; i < addrSize; i += 5) {
204 					addresses = AHAddrLocalServiceUtil.getAHAddrs(i, i + 5);
205 					for (final AHAddr addr : addresses) {
206 						final int size = AHOfferLocalServiceUtil
207 						        .countOfferByAddress(addr.getAddrId());
208 						if (size <= 0) {
209 							AHAddrLocalServiceUtil.deleteAHAddr(addr
210 							        .getAddrId());
211 							m_objLog.info("Removing address "
212 							        + addr.getStreet() + " " + addr.getNumber()
213 							        + " with " + size + " offers!");
214 						}
215 					}
216 				}
217 			}
218 
219 		} catch (final Throwable t) {
220 			m_objLog.warn("Check for orphaned addresses/regions failed due to "
221 			        + t.getMessage(), t);
222 		}
223 	}
224 
225 	/**
226 	 * Removes the orphaned regions.
227 	 */
228 	public void removeOrphanedRegions() {
229 		try {
230 			final int regSize = AHRegionLocalServiceUtil.getAHRegionsCount();
231 			if (regSize > 0) {
232 				List<AHRegion> regions;
233 				for (int i = 0; i < regSize; i += 5) {
234 					regions = AHRegionLocalServiceUtil.getAHRegions(i, i + 5);
235 					for (final AHRegion region : regions) {
236 						// TODO : count only
237 						final List<AHAddr> addrs = AHAddrLocalServiceUtil
238 						        .findAddressesForRegion(region.getRegionId());
239 						if (addrs == null || addrs.size() == 0) {
240 							AHRegionLocalServiceUtil.removeRegion(region
241 							        .getRegionId());
242 							m_objLog.info("Removing region " + region.getZip()
243 							        + ", " + region.getCity()
244 							        + " without addresses!");
245 						}
246 					}
247 				}
248 			}
249 
250 		} catch (final Throwable t) {
251 			m_objLog.warn("Check for orphaned addresses/regions failed due to "
252 			        + t.getMessage(), t);
253 		}
254 	}
