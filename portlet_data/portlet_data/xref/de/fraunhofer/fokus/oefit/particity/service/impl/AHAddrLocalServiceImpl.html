78  	/* (non-Javadoc)
79  	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHAddrLocalService#addAddress(java.lang.String, java.lang.String, java.lang.String, java.lang.String, long)
80  	 */
81  	@Override
82  	public AHAddr addAddress(final String street, final String number,
83  	        Float lat, Float lon, final long regionId) {
84  		AHAddr result = null;
85  
86  		result = this.getAddress(street, number, regionId);
87  		if (result == null) {
88  			try {
89  				result = this.createAHAddr(CounterLocalServiceUtil
90  				        .increment(AHAddr.class.getName()));
91  				result.setStreet(street);
92  				result.setNumber(number);
93  				result.setRegionId(regionId);
94  			} catch (final SystemException e) {
95  				m_objLog.error(e);
96  			}
97  		}
98  		if (lat != null && lon != null) {
99  			result.setCoordLat(lat);
100 			result.setCoordLon(lon);
101 		}
102 		try {
103 			result = this.updateAHAddr(result);
104 		} catch (final SystemException e) {
105 			m_objLog.error(e);
106 		}
107 
108 		return result;
109 	}

127 		List<AHAddr> result = null;
128 		try {
129 			result = this.getAHAddrPersistence().findByregion(regionId);

131 			m_objLog.warn(t);
132 		}
133 		return result;
134 	}

136 	/* (non-Javadoc)
137 	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHAddrLocalService#getAddress(java.lang.String, java.lang.String, long)
138 	 */
139 	@Override
140 	public AHAddr getAddress(final String street, final String number,
141 	        final long regionId) {
142 		AHAddr result = null;
143 
144 		try {
145 			final DynamicQuery dyQuery = DynamicQueryFactoryUtil
146 			        .forClass(AHAddr.class);
147 
148 			final Criterion term1 = RestrictionsFactoryUtil.ilike("street",
149 			        street);
150 			final Criterion term2 = RestrictionsFactoryUtil
151 			        .eq("number", number);
152 			final Criterion term3 = RestrictionsFactoryUtil.eq("regionId",
153 			        regionId);
154 			final Criterion allTerms = RestrictionsFactoryUtil.and(term3,
155 			        RestrictionsFactoryUtil.and(term1, term2));
156 
157 			dyQuery.add(allTerms);
158 			final List addrs = this.dynamicQuery(dyQuery);
159 			if (addrs.size() > 0) {
160 				for (final Object addr : addrs) {
161 					if (addr instanceof AHAddr) {
162 						result = (AHAddr) addr;
163 						break;
164 					}
165 				}
166 			}
167 		} catch (final Throwable t) {
168 			m_objLog.error(t);
169 		}
170 		return result;
171 	}

184 		AHAddr result = null;
185 
186 		try {
187 			result = this.getAHAddr(addrId);
188 		} catch (final Throwable t) {
189 		}
190 		if (result != null) {
191 			try {
192 				result.setCoordLat(lat);
193 				result.setCoordLon(lon);
194 				result = this.updateAHAddr(result);
195 			} catch (final SystemException e) {
196 				m_objLog.error(e);
197 			}
198 		}
199 
200 		return result;
