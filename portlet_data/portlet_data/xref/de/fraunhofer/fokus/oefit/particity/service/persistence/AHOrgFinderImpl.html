34  package de.fraunhofer.fokus.oefit.particity.service.persistence;
35  
36  import java.util.List;
37  
38  import com.liferay.portal.kernel.dao.orm.SQLQuery;
39  import com.liferay.portal.kernel.dao.orm.Session;
40  import com.liferay.portal.kernel.exception.SystemException;
41  import com.liferay.portal.kernel.log.Log;
42  import com.liferay.portal.kernel.log.LogFactoryUtil;
43  import com.liferay.portal.service.persistence.impl.BasePersistenceImpl;
44  
45  import de.fraunhofer.fokus.oefit.adhoc.custom.E_OrderType;
46  import de.fraunhofer.fokus.oefit.particity.model.AHOrg;
47  import de.fraunhofer.fokus.oefit.particity.model.impl.AHOrgImpl;
48  
49  public class AHOrgFinderImpl extends BasePersistenceImpl<AHOrg>
50  		implements AHOrgFinder {
51  	
52  	private static final Log m_objLog = LogFactoryUtil
53  			.getLog(AHOrgFinderImpl.class);
54  
55  	public static String getOrgByUserListEntry = "select * from AHORG org WHERE org.userlist LIKE '%USERMAIL%'";
56  	
57  	public static String getOrgCustomOrder = "select * from AHORG ORDER BY _COL_ _DIR_";
58  	
59  		
60  	@Override
61  	public List<AHOrg> getOrganisationsByUserlistEntry(String userMailAddr)  throws SystemException {
62  		List<AHOrg> result = null;
63  
64  		Session session = null;
65  
66  		try {
67  	
68  			session = openSession();
69  			
70  			SQLQuery query = session.createSQLQuery(getOrgByUserListEntry.replaceAll("USERMAIL", userMailAddr));
71  			
72  			query.addEntity("AHOrg", AHOrgImpl.class);
73  			
74  
75  			result = (List<AHOrg>) query.list();
76  		} catch (Exception e) {
77  			throw new SystemException(e);
78  		} finally {
79  			closeSession(session);
80  		}
81  		return result;
82  	}
83  	
84  	@Override
85  	public List<AHOrg> getOrganisationsWithCustomOrder(String column, String order, int from, int to)
86  			throws SystemException {
87  		List<AHOrg> result = null;
88  
89  		Session session = null;
90  
91  		try {
92  			session = openSession();
93  
94  			String sql = getOrgCustomOrder.replaceAll("_COL_", column).replaceAll("_DIR_", order);
95  			SQLQuery query = session.createSQLQuery(sql);
96  
97  			query.addEntity("AHOrg", AHOrgImpl.class);
98  			query.setFirstResult(from);
99  			query.setMaxResults(to-from);
100 
101 			
102 			result = (List<AHOrg>) query.list();
103 		} catch (Exception e) {
104 			throw new SystemException(e);
105 		} finally {
106 			closeSession(session);
107 		}
108 		return result;
109 	}

111 }
