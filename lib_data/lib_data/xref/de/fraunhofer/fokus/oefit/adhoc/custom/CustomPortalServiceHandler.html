87  	/**
88  	 * Validates a given form object and changes user settings accordingly
89  	 *
90  	 * @param oldUser the old Liferay user model
91  	 * @param newProfile the profile containing new values
92  	 * @param result the message handler to bind validation errors to
93  	 */
94  	public static void changeUserProfile(User oldUser,
95  	        final ProfileForm newProfile, final BindingResult result) {
96  		final String oldMail = oldUser.getEmailAddress();
97  		final String newMail = newProfile.getMail();
98  		if (!oldMail.toLowerCase().trim().equals(newMail.toLowerCase().trim())) {
99  			m_objLog.debug("User mail changed!");
100 			try {
101 				oldUser = UserLocalServiceUtil.updateEmailAddress(
102 				        oldUser.getUserId(), oldUser.getPassword(),
103 				        newMail.trim(), newMail.trim());
104 			} catch (final Throwable e) {
105 				result.rejectValue("mail", "notExistentErrorCode",
106 				        "common.form.profile.field.mail.unknown");
107 				e.printStackTrace();
108 			}
109 			if (oldUser != null) {
110 				AHOrgLocalServiceUtil.updateOwner(oldMail, newMail.trim());
111 			}
112 		} else {
113 			m_objLog.debug("User mail not changed!");
114 		}
115 		if (newProfile.getPass1().equals(newProfile.getPass2())) {
116 			if (newProfile.getPass1().replaceAll("\\*", "").length() > 0) {
117 				m_objLog.debug("User password changed!");
118 				try {
119 					UserLocalServiceUtil.updatePassword(oldUser.getUserId(),
120 					        newProfile.getPass1(),
121 					        newProfile.getPass1(), false);
122 				} catch (final Throwable e) {
123 					result.rejectValue("mail", "notExistentErrorCode",
124 					        "common.form.profile.field.pass1.unknown");
125 					m_objLog.error(e);
126 				}
127 
128 			} else {
129 				m_objLog.debug("User password not changed!");
130 			}
131 		}
132 	}

158 	public static void createOrganisationLogoFolder(long groupId, long userId) throws PortalException, SystemException {
159 		final ServiceContext ctx = new ServiceContext();
160 		ctx.setUserId(userId);
161 		ctx.setScopeGroupId(groupId);
162 		ctx.setAddGroupPermissions(true);
163 		ctx.setAddGuestPermissions(true);
164 		// ctx.setWorkflowAction(WorkflowConstants.ACTION_PUBLISH);
165 
166 		String orgFolder = getConfigValue(E_ConfigKey.ORGANISATION_LOGO_FOLDER);
167 		
168 		Folder parent = null;
169 		try {
170 			parent = DLAppServiceUtil.getFolder(groupId, 0,
171 			        orgFolder);
172 		} catch (final Throwable t) {
173 		}
174 
175 		if (parent == null) {
176 			parent = DLAppServiceUtil.addFolder(groupId, 0,
177 			        orgFolder, "", ctx);
178 			m_objLog.info("Folder "
179 			        + orgFolder
180 			        + " created ....");
181 		} else {
182 			m_objLog.info("Folder "
183 			        + orgFolder
184 			        + " already existing ....");
185 		}
186 
187 		final String[] actionids = new String[2];
188 		final Role guestRole = RoleLocalServiceUtil.getRole(
189 		        PortalUtil.getDefaultCompanyId(), RoleConstants.GUEST);
190 		actionids[0] = ActionKeys.VIEW;
191 		actionids[1] = ActionKeys.ACCESS;
192 		// actionids[2] = ActionKeys.ADD_FILE;
193 
194 		ResourcePermissionServiceUtil.setIndividualResourcePermissions(
195 		        parent.getGroupId(), parent.getCompanyId(),
196 		        DLFolder.class.getName(), parent.getFolderId() + "",
197 		        guestRole.getRoleId(), actionids);
198 		
199 	}

203 	 *
204 	 * @param cmpRole the Liferay role
205 	 * @return the supported enum or null, if not supported
206 	 */
207 	public static E_Role matchesRole(final Role cmpRole) {
208 		E_Role result = null;
209 
210 		if (cmpRole != null) {
211 			for (final E_Role role : E_Role.values()) {
212 				if (getRoleName(role).equals(cmpRole.getName())
213 				        && role.getType() == cmpRole.getType()) {
214 					result = role;
215 					// System.out.println(role.getName()+" ("+role.getType()+") == "+cmpRole.getName()+" ("+cmpRole.getType()+")");
216 					break;
217 				} else {
218 					// System.out.println(role.getName()+" ("+role.getType()+") != "+cmpRole.getName()+" ("+cmpRole.getType()+")");
219 				}
220 			}
221 		}
222 
223 		return result;
224 	}

314 	/**
315 	 * Creates a new portal user
316 	 *
317 	 * @param firstName the first name
318 	 * @param lastName the last name
319 	 * @param mail the email address
320 	 * @param companyId the company id
321 	 * @param locale the user's locale
322 	 * @return the user object
323 	 */
324 	public static User createPortalUser(final String firstName,
325 	        final String lastName, final String mail, final long companyId, final long groupId,
326 	        final Locale locale, boolean sendMail, String password, boolean isAdmin, E_Role defaultRole) {
327 
328 		User user = null;
329 		try {
330 			if (defaultRole == null)
331 				defaultRole = E_Role.ORG;
332 			final Role defRole = checkRole(0, companyId, defaultRole);
333 
334 			try {
335 				user = UserLocalServiceUtil.getUserByEmailAddress(companyId,
336 				        mail);
337 			} catch (final Throwable t) {
338 			}
339 			
340 
341 			if (user == null) {
342 				long[] userGroups = new long[1];
343 				Group guestSite = GroupLocalServiceUtil.getGroup(companyId, GroupConstants.GUEST);
344 				if (guestSite != null) {
345 					userGroups = new long[2];
346 					userGroups[1] = guestSite.getGroupId();
347 				}
348 				userGroups[0] = groupId;
349 				//long[] userGroups = new long[]{groupId, guestSite.getGroupId()};
350 				
351 				long[] roles = new long[2];
352 				roles[0] = defRole.getRoleId();
353 				
354 				if (isAdmin) {
355 					final Role adminRole = checkRole(0, companyId,
356 					        RoleConstants.ADMINISTRATOR, RoleConstants.TYPE_REGULAR);
357 					roles[0] = adminRole.getRoleId();
358 				}
359 				
360 				final Role memberRole = checkRole(0, companyId,
361 				        RoleConstants.SITE_MEMBER, RoleConstants.TYPE_SITE);
362 				roles[1] = memberRole.getRoleId();
363 				
364 				user = UserLocalServiceUtil.addUser(0, companyId, password == null, password,
365 				        password,
366 				        true, null, mail, 0L, "", locale, firstName, "",
367 				        lastName,
368 				        0, 0, false, 0, 1, 1970, "", userGroups, null,
369 				        roles, null, sendMail,
370 				        new ServiceContext());
371 			}
372 
373 		} catch (final Exception e) {
374 			m_objLog.error(e);
375 		}
376 		return user;
377 	}

379 	/**
380 	 * Gets the value for a given configuration key (or the default setting if not found)
381 	 * This method may either access the portlet data-source for local settings or the portal properties for global configuration settings
382 	 *
383 	 * @param key the configuration key
384 	 * @return the configuration value
385 	 */
386 	public static String getConfigValue(final E_ConfigKey key) {
387 		String value = null;
388 		if (key.isSystemProperty()) {
389 			try {
390 				value = PrefsPropsUtil.getString(
391 				        PortalUtil.getDefaultCompanyId(),
392 				        key.getSystemProperty());
393 				// m_objLog.debug("Got portal property "+key.getSystemProperty()+" = "+value);
394 			} catch (final SystemException e) {
395 				value = key.getDefaultValue();
396 				m_objLog.error(e);
397 			}
398 		} else {
399 			value = AHConfigLocalServiceUtil.getConfig(key.toString(),
400 			        key.getDefaultValue());
401 			m_objLog.info("Got portlet property "+key.toString()+" = "+value);
402 		}
403 		return value;
404 	}

457 	/**
458 	 * Save a set of configuration changes right from the portlet request
459 	 * 
460 	 * The request is parsed for configuration values with names out of <code>E_ConfigKey</code>
461 	 * 
462 	 * As a result, social media plugins and messagec composers will be reinitialized to address the changes
463 	 *
464 	 * @param request the portlet request
465 	 */
466 	public static void saveConfig(final ActionRequest request) {
467 		
468 		final Enumeration<String> paramNames = request.getParameterNames();
469 		while (paramNames.hasMoreElements()) {
470 			final String keyName = paramNames.nextElement();
471 			E_ConfigKey key = null;
472 			try {
473 				key = E_ConfigKey.valueOf(keyName);
474 			} catch (final Throwable t) {
475 			}
476 			if (key != null) {
477 
478 				final String value = request.getParameter(keyName);
479 				if (value != null) {
480 					m_objLog.info("Accepting save parameter: "+keyName+" with value: "+value);
481 					final String oldVal = getConfigValue(key);
482 					if (!oldVal.equals(value)) {
483 						m_objLog.debug("Setting changed parameter: " + keyName);
484 						setConfig(key, value);
485 					}
486 
487 				} /*
488 				 * else { m_objLog.debug("Accepting save parameter: "+keyName+
489 				 * " with empty value"); }
490 				 */
491 			} /*
492 			 * else { m_objLog.debug("Ignoring save parameter: "+keyName); }
493 			 */
494 		}
495 		/*
496 		 * refresh modules that depend on fresh configuration settings
497 		 */
498 		MessageComposer.getInstance().refreshCfg();
499 		SocialMediaFactory.refreshCfg();
500 	}

517 	/**
518 	 * Update portal preferences.
519 	 *
520 	 * @param key the property name
521 	 * @param value the value
522 	 */
523 	private static void updatePortalPreferences(final String key,
524 	        final String value) {
525 		try {
526 			final List<PortalPreferences> prefs = PortalPreferencesLocalServiceUtil
527 			        .getPortalPreferenceses(QueryUtil.ALL_POS,
528 			                QueryUtil.ALL_POS);
529 
530 			for (final PortalPreferences pref : prefs) {
531 				final long ownerId = pref.getOwnerId();
532 				final int ownerType = pref.getOwnerType();
533 				// m_objLog.debug("Pref is "+ownerId+","+ownerType+","+pref.getPortalPreferencesId());
534 				final PortletPreferences pps = PortalPreferencesLocalServiceUtil
535 				        .getPreferences(ownerId, ownerType);
536 				final Enumeration<String> names = pps.getNames();
537 				while (names.hasMoreElements()) {
538 					final String pkey = names.nextElement();
539 					if (pkey.equals(key)) {
540 						// m_objLog.debug("Found property "+key+" with former value "+pps.getValue(key,"N/A"));
541 						pps.setValue(pkey, value);
542 						pps.store();
543 						break;
544 					}
545 				}
546 			}
547 		} catch (final Throwable e) {
548 			m_objLog.error(e);
549 		}
