10  public class SchemaVersionHandler extends DefaultHandler {
11  
12  	private Log m_objLog = LogFactoryUtil.getLog(SchemaVersionHandler.class);
13  	
14  	private String m_strVersion = null;
15  	
16  	@Override
17      public void startElement(String uri, String localName, String qName,
18              Attributes attributes) throws SAXException {
19  		
20  	    if (qName.equals("ImportExportRoot")) {
21  	    	//m_objLog.info("Found root elem");
22  	    	if (attributes != null) {
23  	    		for (int i=0; i<attributes.getLength();i++) {
24  	    			//m_objLog.info("Found attribute with local "+attributes.getLocalName(i)+", qName "+attributes.getQName(i));
25  	    			if (attributes.getQName(i).equals("Version")) {
26  	    				//m_objLog.info("Found attributes!");
27  	    				m_strVersion = attributes.getValue(i);
28  	    			}
29  	    		}
30  	    	}
31  	    	// root element has been parsed,
32  	    	// regardless of the result, stop right here
33  	    	throw new SchemaVersionEscapeException();
34  	    }
35      }
36  	
37  	public String getVersion() {
38  		return m_strVersion;
39  	}
