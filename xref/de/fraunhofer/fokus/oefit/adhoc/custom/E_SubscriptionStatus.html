34  package de.fraunhofer.fokus.oefit.adhoc.custom;
35  
36  /**
37   * An enum describing status values available for subscriptions
38   */
39  public enum E_SubscriptionStatus {
40  
41  	/** A status indicating a new (not validated) subscription. */
42  	NEW(1, "enum.subscription.status.new"),
43  	
44  	/** A status indicating a validated subscription. */
45  	VALIDATED(2, "enum.subscription.status.validated");
46  
47  	/**
48  	 * Find by value.
49  	 *
50  	 * @param val the val
51  	 * @return the e_ subscription status
52  	 */
53  	public static E_SubscriptionStatus findByValue(final int val) {
54  		E_SubscriptionStatus result = null;
55  
56  		for (final E_SubscriptionStatus status : E_SubscriptionStatus.values()) {
57  			if (status.getIntValue() == val) {
58  				result = status;
59  				break;
60  			}
61  		}
62  
63  		return result;
64  	}
65  
66  	private String	m_strMsgProperty;
67  
68  	private int	   m_numValue;
69  
70  	private E_SubscriptionStatus(final int value, final String msgProperty) {
71  		this.m_numValue = value;
72  		this.m_strMsgProperty = msgProperty;
73  	}
74  
75  	/**
76  	 * Gets the int value.
77  	 *
78  	 * @return the int value
79  	 */
80  	public int getIntValue() {
81  		return this.m_numValue;
82  	}
83  
84  	/**
85  	 * Gets the msg property.
86  	 *
87  	 * @return the msg property
88  	 */
89  	public String getMsgProperty() {
90  		return this.m_strMsgProperty;
91  	}
92  }
