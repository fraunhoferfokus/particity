34  package de.fraunhofer.fokus.oefit.adhoc.custom;
35  
36  import com.liferay.portal.model.RoleConstants;
37  
38  /**
39   * An enum describing supported roles along with their default locations to redirect to after login.
40   * 
41   * TODO: Should be made configurable in the settings
42   */
43  public enum E_Role {
44  	
45  	/** The null. */
46  	NULL("@NIL@", RoleConstants.TYPE_REGULAR, "/"),
47  	
48  	/** The default site user = guest. */
49  	SITEGUEST(RoleConstants.GUEST, RoleConstants.TYPE_REGULAR, "/"),
50  	
51  	/** The siteadmin. */
52  	SITEADMIN(
53  	        "Administrator",
54  	        RoleConstants.TYPE_REGULAR,
55  	        "/int/administration"),
56  	
57  	/** The admin. */
58  	ADMIN(null, E_ConfigKey.ROLE_NAME_ADM,
59  	        RoleConstants.TYPE_REGULAR,
60  	        "/int/administration"),
61  	
62  	/** The org. */
63  	ORG(E_ConfigKey.ROLE_NAME_ORG,
64  	        RoleConstants.TYPE_REGULAR,
65  	        "/int/organisation"),
66  	
67  	/** The mgmt. */
68  	MGMT(E_ConfigKey.ROLE_NAME_MGMT, RoleConstants.TYPE_REGULAR, "/int/verwaltung"), ;
69  
70  	private E_ConfigKey m_objKey = null;;
71  	private String m_strName = null;
72  	private int	   m_numType;
73  	private String	m_strHomeUrl;
74  
75  	private E_Role(String name, int type, String userHomeUrl) {
76  		this(name, null, type, userHomeUrl);
77  	}
78  	
79  	private E_Role(final E_ConfigKey ckey, int type, String userHomeUrl) {
80  		this(null, ckey, type, userHomeUrl);
81  	}
82  	
83  	private E_Role(String name, final E_ConfigKey key, int type, String userHomeUrl) {
84  		this.m_strName = name;
85  		this.m_objKey = key;
86  		this.m_numType = type;
87  		this.m_strHomeUrl = userHomeUrl;
88  	}
89  
90  	/**
91  	 * Gets the home url.
92  	 *
93  	 * @return the home url
94  	 */
95  	public String getHomeUrl() {
96  		return this.m_strHomeUrl;
97  	}
98  
99  	/**
100 	 * Gets the configuration key
101 	 *
102 	 * @return the config key enum
103 	 */
104 	public E_ConfigKey getKey() {
105 		return this.m_objKey;
106 	}
107 
108 	/**
109 	 * Get the role's default name
110 	 * NOTE: If there is a getKey() referenced, this value might have changed in the system configuration. Use <code>CustomPortalServiceHandler.getRoleName()</code> instead
111 	 * 
112 	 * @return
113 	 */
114 	public String getDefaultName() {
115 		return m_strName;
116 	}
117 	
118 	/**
119 	 * Gets the type.
120 	 *
121 	 * @return the type
122 	 */
123 	public int getType() {
124 		return this.m_numType;
125 	}
126 }
