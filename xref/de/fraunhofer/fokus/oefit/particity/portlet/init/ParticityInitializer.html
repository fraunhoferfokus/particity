230 						false, site.getLayoutId(), path.getThemeId(), "01", "", false);
231 			} else
232 				m_objLog.warn("Did not find theme "+path.getThemeId()+"!");
233             LayoutTemplate template = getLayoutTemplate(path.getTemplateId());
234 			
235 			LayoutTypePortlet layoutTypePortlet = (LayoutTypePortlet) site.getLayoutType();
236 			if (template != null) {
237 				layoutTypePortlet.setLayoutTemplateId(0, path.getTemplateId(), false);
238 			} else
239 				m_objLog.warn("Did not find layout template "+path.getTemplateId()+"!");
240 			updatePermissions(site, path);
241 		} catch (Throwable t) {
242 			m_objLog.error(t);

336 	public static void initSampleContent(long groupId, long adminId, long companyId, Map<E_ContextPath, Layout> layouts, Map<E_ContextPath, String> pathMap) {
337 		Layout layout = null;
338 		Map<String, JournalArticle> articles = new HashMap<String, JournalArticle>();
339 		for (E_SampleContent content: E_SampleContent.values()) {
340 			layout = layouts.get(content.getContext());
341 			if (layout == null) {
342 				m_objLog.debug("Could not find layout for sample content context path "+content.getContext().getPath());
343 				continue;
344 			}
345 			try {
346 				// add portlet to page
347 				if (content.isPorlet()) {
348 					addPortletToPage(layout, content.getDataPath(), content.getContext(), adminId);					
349 					m_objLog.debug("Added sample content "+content.name()+" with portlet "+content.getDataPath()+" for URL "+content.getContext().getPath());
350 				} 
351 				// import HTML as article + add to page
352 				else {
353 					JournalArticle article = articles.get(content.getDataPath());
354 					if (article == null)
355 						article = getArticle(layout.getGroupId(), content.name());
356 					if (article == null) {
357 						InputStream frontend = ParticityInitializer.class.getResourceAsStream(content.getDataPath());
358 						if (frontend != null) {
359 							String contentSrc = IOUtils.toString(frontend);
360 							if (contentSrc != null) {
361 								contentSrc = translatePaths(contentSrc, pathMap);
362 								article = addArticle(adminId, layout.getGroupId(), content.name(), content.getTitle(), contentSrc);
363 								//m_objLog.debug("Added article for "+content.name());
364 								articles.put(content.getDataPath(), article);									
365 							} else {
366 								m_objLog.warn("Could not load sample content "+content.name()+" for URL "+content.getContext().getPath()+" from "+content.getDataPath());
367 							}
368 						}
369 					}
370 					if (article != null) {
371 						layout = layouts.get(content.getContext());
372 						if (layout != null) {
373 							// is nested portlet?
374 							if (content.hasParent()) {
375 								setPortletArticle(content.getParentPortletId(), article.getArticleId(), layout, adminId, layout.getGroupId(), layout.getCompanyId());
376 							} 
377 							// is standalone portlet
378 							else {
379 								// check if article not added yet
380 								if (!checkArticleOnLayout(layout, article.getArticleId(), content.getContext().getColumnId(), layout.getCompanyId())) {
381 									addArticle(content.getContext(), article,layout, adminId, layout.getGroupId(), layout.getCompanyId());
382 									m_objLog.debug("Added sample content "+content.name()+" for URL "+content.getContext().getPath());
383 								} else {
384 									m_objLog.debug("Found existing sample content "+content.name()+" for URL "+content.getContext().getPath());
385 								}
386 							}
387 						} else {
388 							m_objLog.warn("Could not find layout for sample content context path "+content.getContext().getPath());			
389 						}
390 					} else {
391 						m_objLog.warn("Could not create sample content "+content.name()+" for URL "+content.getContext().getPath()+" (already exists?) ");
392 					}
393 				}
394 			} catch (Throwable t) {
395 				m_objLog.warn(t);
396 			}
397 		}
398 	}

411 	public static Map<E_ContextPath, Layout> clearLayouts(long groupId, long adminId, long companyId, Map<E_ContextPath, String> pathMap) {
412 		Map<E_ContextPath, Layout> result = new HashMap<E_ContextPath, Layout>();
413 		
414 		for (E_ContextPath pth: E_ContextPath.values()) {
415 			String actualPath = pathMap.get(pth);
416 			if (actualPath == null)
417 				actualPath = pth.getPath();
418 			Layout layout = getLayout(groupId, actualPath);
419 			// if no layout at new path, try the original path setting
420 			if (layout == null && !pth.getPath().equals(actualPath))
421 				layout = getLayout(groupId, pth.getPath());
422 			if (layout == null) {
423 				layout = createLayout(adminId, companyId, groupId, pth, actualPath);
424 				m_objLog.debug("Created layout for URL "+actualPath);
425 			} else {
426 				m_objLog.debug("Found layout for URL "+actualPath);
427 				clearPage(layout, pth, actualPath);
428 			}
429 			result.put(pth, layout);
430 		}
431 		return result;
432 	}

466 			result = LayoutLocalServiceUtil.getFriendlyURLLayout(groupId,

469 					+ " and url " + context);

471 			m_objLog.warn("No layout registered for group " + groupId
472 					+ " with url " + context);
473 			// retry searching manually
474 			if (result == null) {
475 				try {
476 					List<Layout> layouts = LayoutLocalServiceUtil.getLayouts(QueryUtil.ALL_POS, QueryUtil.ALL_POS);
477 					for (Layout layout: layouts) {
478 						if (layout.isPublicLayout() && layout.getFriendlyURL().equals(context)) {
479 							result = layout;
480 							m_objLog.debug("Found matching public layout "+context+" with group "+result.getGroupId()+", company "+result.getCompanyId());
481 							break;
482 						}
483 					}
484 				} catch (Throwable t2) {
485 					t2.printStackTrace();
486 				}
487 			}
