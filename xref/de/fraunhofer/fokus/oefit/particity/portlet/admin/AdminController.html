165 	@RequestMapping(value = "view")
166 	@ActionMapping(params = "action=exportDatabase")
167 	public void exportDb(final ActionRequest request, final ActionResponse response,
168 	        final Model model) {
169 		m_objLog.debug("exportDatabase::start");
170 		
171 		ByteArrayOutputStream bout = null;
172 		//GZIPOutputStream zipout = null;
173 		
174 		try {
175 			HttpServletRequest httpServletRequest = (HttpServletRequest) PortalUtil
176 					.getHttpServletRequest(request);
177 			HttpServletResponse httpServletResponse = (HttpServletResponse) PortalUtil
178 					.getHttpServletResponse(response);
179 	
180 			httpServletResponse.setHeader("Expires", "0");
181 			httpServletResponse.setHeader("Cache-Control",
182 					"must-revalidate, post-check=0, pre-check=0");
183 			httpServletResponse.setHeader("Pragma", "public");
184 			
185 			
186 			ExportWriter export = new ExportWriter(getCompanyId(request));
187 			
188 			SimpleDateFormat sdf = new SimpleDateFormat("yyyy-dd-MM");
189 			String dateStr = sdf.format(Calendar.getInstance().getTime());
190 			String lrHome = PropsUtil.get(PropsKeys.LIFERAY_HOME);
191 			
192 			boolean toFileDone = false;
193 			// write to file
194 			if (lrHome != null) {
195 				File lrBackupDir = new File(lrHome+File.separator+"backup");
196 				if (!lrBackupDir.exists()) {
197 					lrBackupDir.mkdir();
198 				}
199 				if (lrBackupDir.exists()) {
200 					File backupFile = new File(lrBackupDir,"particity_"+dateStr+".xml");
201 					if (backupFile.exists())
202 						toFileDone = backupFile.delete();
203 					if (!backupFile.exists()) {
204 						toFileDone = backupFile.createNewFile();
205 						if (toFileDone) {
206 							m_objLog.debug("Writing out to "+backupFile.getAbsolutePath());
207 							FileOutputStream fout = new FileOutputStream(backupFile);
208 							export.writeToStream(fout);
209 							fout.flush();
210 							fout.close();
211 							//SessionMessages.add(request, "request_processed", "admin.jsp.db.export.success");
212 							response.setRenderParameter("exportFilename", backupFile.getAbsolutePath());
213 							Map<String, String> logs = export.getLog();
214 							export.cleanup();
215 							request.setAttribute("exportLogs", logs);
216 						} else {
217 							m_objLog.warn("Could not create backup file at "+backupFile.getAbsolutePath());
218 						}
219 					}
220 				} else {
221 					m_objLog.warn("Could not create backup directory at "+lrBackupDir.getAbsolutePath());
222 				}
223 				
224 			} else {
225 				m_objLog.warn("Could not evaluate liferay home");
226 			}
227 			// if for any reason the file could not be written, try to send it to the client
228 			if (!toFileDone) {
229 				// write to client
230 				bout = new ByteArrayOutputStream();
231 				export.writeToStream(bout);
232 				bout.flush();
233 					
234 				com.liferay.portal.kernel.servlet.ServletResponseUtil.sendFile(
235 						httpServletRequest, httpServletResponse, "particity_"+dateStr+".xml", bout.toByteArray());
236 				Map<String, String> logs = export.getLog();
237 				export.cleanup();
238 				request.setAttribute("exportLogs", logs);
239 			}
240 		} catch (Throwable t) {
241 			m_objLog.error("Error exporting data",t);
242 		} finally {
243 			if (bout != null) {
244 				try {
245 	                bout.close();
246                 } catch (IOException e) {
247                 }
248 			}
249 		}
250 		
251 		response.setRenderParameter("tab", "dbtools");
252 		m_objLog.debug("exportDatabase::end");
253 	}

274 	@RequestMapping(value = "view")
275 	@ActionMapping(params = "action=importDatabase")
276 	public void importDatabase(ActionRequest request, ActionResponse response) {
277 	  m_objLog.debug("importDatabase::start");
278 	  UploadPortletRequest uploadRequest = PortalUtil.getUploadPortletRequest(request);
279 	  //Enumeration<?> paramEnum = uploadRequest.getParameterNames();
280 	  //File tempFile = uploadRequest.getFile("file");
281 	  if (uploadRequest.getSize("file") > 0) {
282 		  try {
283 			  InputStream fin = uploadRequest.getFileAsStream("file", true);
284 			  if (fin != null) {
285 				  Map<String,String> logs = ImporterFactory.importData(fin, getCompanyId(request), getGroupId(request), getUserId(request));
286 				  if (logs != null)
287 					  request.setAttribute("importLogs", logs);
288 			  }
289 		  } catch (ImportFailedException t) {
290 			  response.setRenderParameter("errorCode", t.getMessageCode());
291 			  //SessionErrors.add(request, t.getMessageCode());
292 			  m_objLog.warn("Import failed",t);
293 		  } catch (Throwable t) {
294 			  m_objLog.warn("Import failed",t);
295 			  //response.setRenderParameter("errorCode", ImportFailedException.);
296 		  }
297 		  
298 	  } 
299 	  
300 	  response.setRenderParameter("tab", "dbtools");
301 	  m_objLog.debug("importDatabase::end");
302 	}

315 		if (Constants.RESTRICT_TO_DEMO) {
316 			m_objLog.debug("removeCategory::denied()");
317 			SessionErrors.add(request, "common.demo.denied");
318 			return;
319 		}
320 		
321 		m_objLog.debug("removeCategory::start");
322 
323 		final String catId = request.getParameter("catId");
324 
325 		if (catId != null) {
326 			response.setRenderParameter("catId", catId);
327 			final Long l_catId = Long.parseLong(catId);

329 				CustomCategoryServiceHandler.deleteMainCategory(l_catId);

369 	/**
370 	 * Render.
371 	 *
372 	 * @param request the request
373 	 * @param response the response
374 	 * @param model the model
375 	 * @return the string
376 	 */
377 	@RequestMapping(value = "view")
378 	@RenderMapping
379 	public String render(final RenderRequest request,
380 	        final RenderResponse response, final Model model) {
381 		m_objLog.trace("render::start");
382 		
383 		// get IDs of current session
384 		long groupId = getGroupId(request);
385 		long userId = getUserId(request);
386 		
387 		// check whether portal already initialized
388 		CustomPortalServiceHandler.checkInit(groupId, userId);
389 		
390 		String page = request.getParameter("jspPage");
391 		if (page == null) {
392 			page = "admin";
393 		}
394 		final String error = request.getParameter("error");
395 		if (error != null) {
396 			model.addAttribute("error", error);
397 			m_objLog.warn("Handing over error " + error);
398 			final String ePage = request.getParameter("errorPage");
399 			if (ePage != null) {
400 				page = ePage;
401 			}
402 		} /*
403 		 * else { model.addAttribute("data", new BerliosProject()); }
404 		 */
405 		m_objLog.trace("render::end(" + page + ")");
406 		return page;
407 	}
