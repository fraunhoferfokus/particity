72   * @see de.fraunhofer.fokus.oefit.particity.service.AHOrgLocalServiceUtil
73   */
74  public class AHOrgLocalServiceImpl extends AHOrgLocalServiceBaseImpl {
75  	/*
76  	 * NOTE FOR DEVELOPERS: Never reference this interface directly. Always use
77  	 * {@link de.fraunhofer.fokus.oefit.adhoc.service.AHOrgLocalServiceUtil} to
78  	 * access the a h org local service.

80  	private static final Log	m_objLog	= LogFactoryUtil

83  	/* (non-Javadoc)
84  	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHOrgLocalService#addOrganisation(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.lang.String, long, long)
85  	 */
86  	@Override
87  	public AHOrg addOrganisation(final String owner, final String name,
88  	        final String holder, final String descr, final String legalStatus,
89  	        final long addressId, final long contactId) {
90  		m_objLog.trace("addOrganisation::start("+owner+")");
91  		AHOrg result = null;
92  
93  		try {
94  			AHOrg tmp = this.getOrganisationByOwnerMail(owner);
95  			if (tmp == null) {
96  				m_objLog.info("Found no organisation for " + owner
97  				        + ". Creating new one!");
98  				tmp = this.createAHOrg(CounterLocalServiceUtil
99  				        .increment(AHOrg.class.getName()));
100 				tmp.setCreated(System.currentTimeMillis());
101 				tmp.setOwner(owner.toLowerCase());
102 				tmp.setStatus(E_OrgStatus.NEW.getIntValue());
103 			} else {
104 				m_objLog.info("Found existing organisation for " + owner + "!");
105 				tmp.setStatus(E_OrgStatus.CHANGED.getIntValue());
106 			}
107 			tmp.setAddressId(addressId);
108 			tmp.setContactId(contactId);
109 			tmp.setName(name);
110 			tmp.setDescription(descr);
111 			tmp.setUpdated(System.currentTimeMillis());
112 			tmp.setHolder(holder);
113 			tmp.setLegalStatus(legalStatus);
114 			// set valid, if it does not require additional input or check by mgmt and moderation is disabled
115 			if (!CustomPortalServiceHandler.isConfigEnabled(E_ConfigKey.MODERATE_ORGS)) {
116 				tmp.setStatus(E_OrgStatus.VALIDATED.getIntValue());
117 			}
118 			result = this.updateAHOrg(tmp);
119 		} catch (final SystemException e) {
120 			m_objLog.error(e);
121 		}
122 		m_objLog.trace("addOrganisation::end("+owner+")");
123 		return result;
124 	}

131 		try {
132 			final AHOrg org = this.getAHOrg(orgId);
133 			String users = org.getUserlist();
134 			if (users.length() > 0) {
135 				users += ",";
136 			}
137 			users += userMail.toLowerCase();
138 			org.setUserlist(users);
139 			this.updateAHOrg(org);
140 		} catch (final Throwable t) {
141 			m_objLog.error(t);
142 		}
143 	}

295 	@Override
296 	public void updateOwner(final String oldOwner, final String newOwner) {
297 		List<AHOrg> orgs;
298 		try {
299 			orgs = this.getAHOrgPersistence().findByowner(oldOwner);
300 			if (orgs != null && orgs.size() > 0) {
301 				for (final AHOrg org : orgs) {
302 					org.setOwner(newOwner.toLowerCase());
303 					this.updateAHOrg(org);
304 				}
305 			}
306 		} catch (final SystemException e) {
307 			m_objLog.error(e);
