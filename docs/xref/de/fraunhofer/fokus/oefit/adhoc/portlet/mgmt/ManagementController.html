96  	/**
97  	 * Approve offer.
98  	 *
99  	 * @param data the data
100 	 * @param result the result
101 	 * @param request the request
102 	 * @param response the response
103 	 * @param model the model
104 	 */
105 	@ActionMapping(params = "action=approveOffer")
106 	public void approveOffer(
107 	        @Valid @ModelAttribute("data") final OfferForm data,
108 	        final BindingResult result, final ActionRequest request,
109 	        final ActionResponse response, final Model model) {
110 		m_objLog.debug("approveOffer::start(" + data.getOfferId() + ")");
111 
112 		this.m_objOfferFormValidator.validate(data, result);
113 
114 		if (result.hasErrors()) {
115 			response.setRenderParameter("jspPage", "../shared/offer");
116 			response.setRenderParameter("actionType", "approve");
117 		} else {
118 			if (data.getOfferId() >= 0) {
119 
120 				final ThemeDisplay themeDisplay = (ThemeDisplay) request
121 				        .getAttribute(WebKeys.THEME_DISPLAY);
122 
123 				if (!CustomLockServiceHandler.isLocked(AHOffer.class.getName(),
124 				        data.getOfferId(), themeDisplay)) {
125 					try {
126 						if (data.isRequireAgencyContact()) {
127 							m_objLog.debug("Saving with agency contact!");
128 							final AHContact contact = CustomPersistanceServiceHandler
129 							        .addContact(data.getContactForename(),
130 							                data.getContactSurname(),
131 							                data.getContactMail(),
132 							                data.getContactTel());
133 							AHOfferLocalServiceUtil.setSndContact(
134 							        data.getOfferId(), contact.getContactId(),
135 							        E_OfferStatus.VALIDATED);
136 						} else {
137 							m_objLog.debug("Saving with orga contact!");
138 							AHOfferLocalServiceUtil.setOfferStatus(
139 							        data.getOfferId(), E_OfferStatus.VALIDATED);
140 						}
141 						// force unlock
142 						CustomLockServiceHandler.unlock(
143 						        AHOffer.class.getName(), data.getOfferId(),
144 						        themeDisplay);
145 					} catch (final Throwable e) {
146 						m_objLog.error(e);
147 					}
148 				}
149 			}
150 			// redirect to offer tab
151 			response.setRenderParameter("tabId", "offer");
152 		}
153 
154 		m_objLog.debug("approveOffer::end");
155 	}

341 	@ActionMapping(params = "action=publishSocial")
342 	private void publishSocial(final ActionRequest request,
343 	        final ActionResponse response, final Model model) {
344 		final Long offerId = this.getOfferId(request);
345 		final String strType = request.getParameter("type");
346 		m_objLog.debug("publishSocial::start(" + offerId + "," + strType + ")");
347 
348 		this.copyRenderParameter("tabId", request, response);
349 		this.copyRenderParameter("page", request, response);
350 		this.copyRenderParameter("column", request, response);
351 		this.copyRenderParameter("order", request, response);
352 
353 		E_SocialMediaPlugins smtype = null;
354 		if (strType != null) {
355 			try {
356 				smtype = E_SocialMediaPlugins.valueOf(strType);
357 			} catch (final Throwable t) {
358 			}
359 			if (smtype != null) {
360 				final I_SocialMediaClient client = SocialMediaFactory
361 				        .getClient(smtype);
362 				if (client != null) {
363 					final String offerUrlStr = CustomPortalServiceHandler
364 					        .getConfigValue(E_ConfigKey.MGMT_PATH_OFFER_PERMALINK)
365 					        + offerId;
366 					URL offerUrl = null;
367 					try {
368 						offerUrl = new URL(offerUrlStr);
369 					} catch (final Throwable t) {
370 						m_objLog.error("Could not generate offer URL "
371 						        + offerUrlStr);
372 					}
373 					if (offerUrl != null) {
374 						final boolean sent = client.publishOffer(offerUrl,
375 						        offerId);
376 						if (sent) {
377 							AHOfferLocalServiceUtil.addSocialStatus(offerId,
378 							        smtype);
379 						} else {
380 							SessionErrors.add(request,
381 							        "common.socialize.plugin.publishfail");
382 						}
383 					}
384 				}
385 			}
386 		}
387 
388 		m_objLog.debug("publishSocial::end()");
389 	}

399 	@RequestMapping(value = "view")
400 	@RenderMapping
401 	public String render(final RenderRequest request,
402 	        final RenderResponse response, final Model model) {
403 		m_objLog.trace("render::start");
404 		String page = request.getParameter("jspPage");
405 		if (page == null) {
406 			page = "mgmt";
407 		}
408 		final String error = request.getParameter("error");
409 		if (error != null) {
410 			model.addAttribute("error", error);
411 			m_objLog.warn("Handing over error " + error);
412 			final String ePage = request.getParameter("errorPage");
413 			if (ePage != null) {
414 				page = ePage;
415 			}
416 		} /*
417 		 * else { model.addAttribute("data", new BerliosProject()); }
418 		 */
419 		/*
420 		 * for (E_ConfigCategory cat: E_ConfigCategory.values()) { if
421 		 * (cat.getRole().equals(E_Role.MGMT)) {
422 		 * model.addAttribute(cat.toString().toLowerCase()+"CfgData",
423 		 * CustomServiceHandler.getConfig(cat)); } }
424 		 */
425 		m_objLog.trace("render::end(" + page + ")");
426 		return page;
427 	}
