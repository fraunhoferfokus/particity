20  package de.fraunhofer.fokus.oefit.adhoc.portlet.org.registration;
21  
22  import org.springframework.stereotype.Component;
23  import org.springframework.validation.Errors;
24  import org.springframework.validation.Validator;
25  
26  import com.liferay.portal.theme.ThemeDisplay;
27  
28  import de.fraunhofer.fokus.oefit.adhoc.custom.CustomPortalServiceHandler;
29  import de.fraunhofer.fokus.oefit.adhoc.forms.RegistrationForm;
30  import de.fraunhofer.fokus.oefit.adhoc.model.AHOrg;
31  import de.fraunhofer.fokus.oefit.adhoc.service.AHOrgLocalServiceUtil;
32  
33  // TODO: Auto-generated Javadoc
34  /**
35   * The Class RegistrationFormValidator.
36   */
37  @Component
38  public class RegistrationFormValidator implements Validator {
39  
40  	private ThemeDisplay	m_objDisplay;
41  
42  	/**
43  	 * Sets the theme display.
44  	 *
45  	 * @param display the new theme display
46  	 */
47  	public void setThemeDisplay(final ThemeDisplay display) {
48  		this.m_objDisplay = display;
49  	}
50  
51  	/* (non-Javadoc)
52  	 * @see org.springframework.validation.Validator#supports(java.lang.Class)
53  	 */
54  	@Override
55  	public boolean supports(final Class<?> clazz) {
56  		return RegistrationForm.class.equals(clazz);
57  	}
58  
59  	/* (non-Javadoc)
60  	 * @see org.springframework.validation.Validator#validate(java.lang.Object, org.springframework.validation.Errors)
61  	 */
62  	@Override
63  	public void validate(final Object target, final Errors errors) {
64  		final RegistrationForm form = (RegistrationForm) target;
65  
66  		final boolean userExists = CustomPortalServiceHandler.hasUser(
67  		        this.m_objDisplay.getCompanyId(), form.getMail());
68  		final AHOrg org = AHOrgLocalServiceUtil.getOrganisationByOwnerMail(form
69  		        .getMail());
70  
71  		// if the user has already an organisation registered, fail
72  		if (userExists && org != null) {
73  			errors.rejectValue("mail", "notExistentErrorCode",
74  			        "org.form.addOrg.field.mail.exists");
75  		}
76  	}
77  
78  }
