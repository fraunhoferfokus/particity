52  /**
53   * The implementation of the a h region local service.
54   *
55   * <p>
56   * All custom service methods should be put in this class. Whenever methods are
57   * added, rerun ServiceBuilder to copy their definitions into the
58   * {@link de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService}
59   * interface.
60   *
61   * <p>
62   * This is a local service. Methods of this service will not have security
63   * checks based on the propagated JAAS credentials because this service can only
64   * be accessed from within the same VM.
65   * </p>
66   *
67   * @author Brian Wing Shun Chan
68   * @see de.fraunhofer.fokus.oefit.adhoc.service.base.AHRegionLocalServiceBaseImpl
69   * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalServiceUtil
70   */
71  public class AHRegionLocalServiceImpl extends AHRegionLocalServiceBaseImpl {
72  	/*
73  	 * NOTE FOR DEVELOPERS: Never reference this interface directly. Always use
74  	 * {@link de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalServiceUtil}
75  	 * to access the a h region local service.
76  	 */
77  
78  	private static final Log	m_objLog	= LogFactoryUtil
79  	                                             .getLog(AHRegionLocalServiceImpl.class);
80  
81  	/* (non-Javadoc)
82  	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService#addRegion(java.lang.String, java.lang.String, int)
83  	 */
84  	@Override
85  	public AHRegion addRegion(final String city, final String country,
86  	        final int zip) {
87  		AHRegion result = null;
88  
89  		result = this.getRegion(city, country, zip);
90  		if (result == null) {
91  			try {
92  				result = this.createAHRegion(new AHRegionPK(
93  				        CounterLocalServiceUtil.increment(AHRegion.class
94  				                .getName()), zip, city, country));
95  				result = this.updateAHRegion(result);
96  			} catch (final SystemException e) {
97  				m_objLog.error(e);
98  			}
99  		}
100 
101 		return result;
102 	}
103 
104 	/* (non-Javadoc)
105 	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService#getRegion(long)
106 	 */
107 	@Override
108 	public AHRegion getRegion(final long regionId) {
109 		AHRegion result = null;
110 
111 		try {
112 			final List<AHRegion> regions = this.getAHRegionPersistence()
113 			        .findByregionId(
114 			                regionId);
115 			if (regions.size() > 0) {
116 				result = regions.get(0);
117 
118 			}
119 		} catch (final Throwable t) {
120 			m_objLog.error(t);
121 		}
122 
123 		return result;
124 	}
125 
126 	/* (non-Javadoc)
127 	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService#getRegion(java.lang.String, java.lang.String, int)
128 	 */
129 	@Override
130 	public AHRegion getRegion(final String city, final String country,
131 	        final int zip) {
132 		AHRegion result = null;
133 
134 		try {
135 			final DynamicQuery dyQuery = DynamicQueryFactoryUtil
136 			        .forClass(AHRegion.class);
137 
138 			final Criterion term1 = RestrictionsFactoryUtil.ilike(
139 			        "primaryKey.city", city);
140 			final Criterion term2 = RestrictionsFactoryUtil.ilike(
141 			        "primaryKey.country", country);
142 			final Criterion term3 = RestrictionsFactoryUtil.eq(
143 			        "primaryKey.zip", zip);
144 			final Criterion allTerms = RestrictionsFactoryUtil.and(term3,
145 			        RestrictionsFactoryUtil.and(term1, term2));
146 
147 			dyQuery.add(allTerms);
148 			final List regions = this.dynamicQuery(dyQuery);
149 
150 			if (regions.size() > 0) {
151 				for (final Object region : regions) {
152 					if (region instanceof AHRegion) {
153 						result = (AHRegion) region;
154 						break;
155 					}
156 				}
157 			}
158 		} catch (final SystemException e) {
159 			m_objLog.error(e);
160 		}
161 
162 		return result;
163 	}
164 
165 	/* (non-Javadoc)
166 	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService#removeRegion(long)
167 	 */
168 	@Override
169 	public AHRegion removeRegion(final long regionId) {
170 		AHRegion result = this.getRegion(regionId);
171 		if (result != null) {
172 			try {
173 				result = this.deleteAHRegion(result);
174 			} catch (final SystemException e) {
175 				m_objLog.error(e);
176 			}
177 		}
178 		return result;
179 	}

181 	/* (non-Javadoc)
182 	 * @see de.fraunhofer.fokus.oefit.adhoc.service.AHRegionLocalService#removeRegion(java.lang.String, java.lang.String, int)
183 	 */
184 	@Override
185 	public AHRegion removeRegion(final String city, final String country,
186 	        final int zip) {
187 		AHRegion result = null;
188 
189 		final AHRegion region = this.getRegion(city, country, zip);
190 		if (region != null) {
191 			try {
192 				result = this.deleteAHRegion(region);
193 			} catch (final SystemException e) {
194 				m_objLog.error(e);
195 			}
196 		}
197 
198 		return result;
199 	}
