20  package de.fraunhofer.fokus.oefit.adhoc.custom;
21  
22  import com.liferay.portal.kernel.portlet.DefaultFriendlyURLMapper;
23  import com.liferay.portal.kernel.portlet.FriendlyURLMapper;
24  import com.liferay.portal.kernel.portlet.LiferayPortletURL;
25  
26  // TODO: Auto-generated Javadoc
27  /**
28   * The Class CustomURLMapper.
29   */
30  public class CustomURLMapper extends DefaultFriendlyURLMapper implements
31          FriendlyURLMapper {
32  
33  	/* (non-Javadoc)
34  	 * @see com.liferay.portal.kernel.portlet.DefaultFriendlyURLMapper#buildPath(com.liferay.portal.kernel.portlet.LiferayPortletURL)
35  	 */
36  	@Override
37  	public String buildPath(final LiferayPortletURL liferayPortletURL) {
38  		String result = super.buildPath(liferayPortletURL);
39  
40  		if (result != null && result.contains("/-/")) {
41  			result = result.replaceAll("/-/", "/");
42  		}
43  
44  		return result;
45  	}
46  
47  	/* (non-Javadoc)
48  	 * @see com.liferay.portal.kernel.portlet.BaseFriendlyURLMapper#isCheckMappingWithPrefix()
49  	 */
50  	@Override
51  	public boolean isCheckMappingWithPrefix() {
52  
53  		// return false;
54  		return true;
55  	}
56  }
