40  	/**
41  	 * Find by value.
42  	 *
43  	 * @param val the val
44  	 * @return the e_ offer status
45  	 */
46  	public static E_OfferStatus findByValue(final int val) {
47  		E_OfferStatus result = null;
48  
49  		for (final E_OfferStatus status : E_OfferStatus.values()) {
50  			if (status.getIntValue() == val) {
51  				result = status;
52  				break;
53  			}
54  		}
55  
56  		return result;
57  	}
