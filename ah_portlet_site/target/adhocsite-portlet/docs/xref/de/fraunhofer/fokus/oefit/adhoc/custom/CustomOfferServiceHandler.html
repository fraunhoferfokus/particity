55  	/**
56  	 * Adds the offer.
57  	 *
58  	 * @param data the data
59  	 * @return the AH offer
60  	 */
61  	public static AHOffer addOffer(final OfferForm data) {
62  		AHOffer result = null;
63  
64  		String countryName = data.getRegionCountry();
65  
66  		try {
67  			final Long countryId = Long.parseLong(countryName);
68  			final AHCategories countryCat = AHCategoriesLocalServiceUtil
69  			        .getCategory(countryId);
70  			if (countryCat != null) {
71  				countryName = countryCat.getName();
72  			}
73  		} catch (final Throwable t) {
74  			// m_objLog.warn(t);
75  		}
76  		String workHours = data.getWorkHours();
77  
78  		try {
79  			final Long id = Long.parseLong(workHours);
80  			final AHCategories cat = AHCategoriesLocalServiceUtil
81  			        .getCategory(id);
82  			if (cat != null) {
83  				workHours = cat.getName();
84  			}
85  		} catch (final Throwable t) {
86  			// m_objLog.warn(t);
87  		}
88  
89  		final AHRegion region = AHRegionLocalServiceUtil.addRegion(
90  		        data.getRegionCity(), data.getRegionCountry(),
91  		        Integer.parseInt(data.getRegionZip()));
92  		m_objLog.debug("Added/got region " + region.getRegionId());
93  		final AHAddr address = AHAddrLocalServiceUtil.addAddress(
94  		        data.getAddrStreet(), data.getAddrNum(), data.getAddrLat(),
95  		        data.getAddrLon(), region.getRegionId());
96  		m_objLog.debug("Added/got addr " + address.getAddrId());
97  		final AHContact contact = AHContactLocalServiceUtil.addContact(
98  		        data.getContactForename(), data.getContactSurname(),
99  		        data.getContactTel(), null, data.getContactMail(), null);
100 		final AHContact contact2 = AHContactLocalServiceUtil.addContact(
101 		        data.getContactSndForename(), data.getContactSndSurname(),
102 		        data.getContactSndTel(), null, data.getContactSndMail(), null);
103 		m_objLog.debug("Added/got contact " + contact.getContactId());
104 
105 		final E_OfferType type = E_OfferType.valueOf(data.getType());
106 
107 		AHOrg org = null;
108 		try {
109 			org = AHOrgLocalServiceUtil.getAHOrg(data.getOrgId());
110 		} catch (final Throwable t) {
111 			m_objLog.error(t);
112 		}
113 
114 		E_OfferWorkType workType = null;
115 		try {
116 			workType = E_OfferWorkType.valueOf(data.getWorkType());
117 		} catch (final Throwable t) {
118 		}
119 
120 		String[] categories = data.getCategories();
121 		final String[] services = data.getServices();
122 		categories = (String[]) ArrayUtils.addAll(categories, services);
123 
124 		final long[] l_cats = CustomServiceUtils.categoryStrToLong(categories);
125 
126 		final long publishDate = CustomServiceUtils.parseZoneDateTime(
127 		        data.getPublishDate(),
128 		        data.getPublishTime());
129 		final long expireDate = CustomServiceUtils.parseZoneDateTime(
130 		        data.getExpireDate(),
131 		        data.getExpireTime());
132 
133 		if (org != null && region != null && address != null && contact != null
134 		        && workType != null) {
135 			result = AHOfferLocalServiceUtil.addOffer(data.getOfferId(), type,
136 			        data.getTitle(),
137 			        data.getDescr(), workHours, workType,
138 			        publishDate, expireDate,
139 			        address.getAddrId(), contact.getContactId(),
140 			        contact2.getContactId(), data.isRequireAgencyContact(),
141 			        data.getOrgId(), l_cats);
142 		} else {
143 			m_objLog.error("Unexpected values while adding/updating offer "
144 			        + data.getOfferId());
145 		}
146 
147 		return result;
148 	}

150 	/**
151 	 * Gets the offer.
152 	 *
153 	 * @param offer the offer
154 	 * @return the offer
155 	 */
156 	public static OfferForm getOffer(final AHOffer offer) {
157 		OfferForm result = null;
158 
159 		if (offer != null) {
160 			result = new OfferForm();
161 			result.clear();
162 
163 			result.setTitle(offer.getTitle());
164 			result.setDescr(offer.getDescription());
165 			result.setExpireDate(CustomServiceUtils.formatZoneDate(offer
166 			        .getExpires()));
167 			result.setExpireTime(CustomServiceUtils.formatZoneTime(offer
168 			        .getExpires()));
169 			result.setPublishDate(CustomServiceUtils.formatZoneDate(offer
170 			        .getPublish()));
171 			result.setPublishTime(CustomServiceUtils.formatZoneTime(offer
172 			        .getPublish()));
173 			result.setType(E_OfferType.findByValue(offer.getType()).toString());
174 			result.setWorkType(E_OfferWorkType.findByValue(offer.getWorkType())
175 			        .toString());
176 			result.setWorkHours(offer.getWorkTime());
177 			result.setOfferId(offer.getOfferId());
178 			result.setOrgId(offer.getOrgId());
179 			result.setRequireAgencyContact(offer.isContactAgency());
180 
181 			AHAddr addr = null;
182 			AHContact contact = null;
183 			AHContact contact2 = null;
184 			AHRegion region = null;
185 			try {
186 				contact = AHContactLocalServiceUtil.getAHContact(offer
187 				        .getContactId());
188 			} catch (final Throwable t) {
189 				m_objLog.warn(t);
190 			}
191 			try {
192 				contact2 = AHContactLocalServiceUtil.getAHContact(offer
193 				        .getSndContactId());
194 			} catch (final Throwable t) {
195 				m_objLog.warn(t);
196 			}
197 			try {
198 				addr = AHAddrLocalServiceUtil.getAHAddr(offer.getAdressId());
199 			} catch (final Throwable t) {
200 				m_objLog.warn(t);
201 			}
202 			try {
203 				if (addr != null) {
204 					region = AHRegionLocalServiceUtil.getRegion(addr
205 					        .getRegionId());
206 				} else {
207 					m_objLog.warn("No region for address " + addr.getAddrId());
208 				}
209 			} catch (final Throwable t) {
210 				m_objLog.warn(t);
211 			}
212 
213 			if (contact != null) {
214 				result.setContactForename(contact.getForename());
215 				result.setContactSurname(contact.getSurname());
216 				result.setContactMail(contact.getEmail());
217 				result.setContactTel(contact.getTel());
218 			} else {
219 				m_objLog.warn("No contact for offer " + offer.getOfferId());
220 			}
221 			if (contact2 != null) {
222 				result.setContactSndForename(contact2.getForename());
223 				result.setContactSndSurname(contact2.getSurname());
224 				result.setContactSndMail(contact2.getEmail());
225 				result.setContactSndTel(contact2.getTel());
226 			} else {
227 				m_objLog.warn("No 2nd contact for offer " + offer.getOfferId());
228 			}
229 			if (region != null) {
230 				result.setRegionCity(region.getCity());
231 				result.setRegionCountry(region.getCountry());
232 				result.setRegionZip(Integer.toString(region.getZip()));
233 			} else {
234 				m_objLog.warn("No region for offer " + offer.getOfferId());
235 			}
236 			if (addr != null) {
237 				result.setAddrNum(addr.getNumber());
238 				result.setAddrStreet(addr.getStreet());
239 				result.setAddrLat(addr.getCoordLat());
240 				result.setAddrLon(addr.getCoordLon());
241 			} else {
242 				m_objLog.warn("No address for offer " + offer.getOfferId());
243 			}
244 
245 			List<AHCatEntries> categories = AHOfferLocalServiceUtil
246 			        .getCategoriesByOffer(offer.getOfferId(),
247 			                E_CategoryType.SEARCH);
248 			if (categories != null && categories.size() > 0) {
249 				final String[] catArr = new String[categories.size()];
250 				for (int i = 0; i < categories.size(); i++) {
251 					catArr[i] = Long.toString(categories.get(i).getItemId());
252 				}
253 				result.setCategories(catArr);
254 			}
255 
256 			categories = AHOfferLocalServiceUtil.getCategoriesByOffer(
257 			        offer.getOfferId(), E_CategoryType.OFFERCATS);
258 			if (categories != null && categories.size() > 0) {
259 				final String[] catArr = new String[categories.size()];
260 				for (int i = 0; i < categories.size(); i++) {
261 					catArr[i] = Long.toString(categories.get(i).getItemId());
262 				}
263 				result.setServices(catArr);
264 			}
265 		}
266 
267 		return result;
268 	}

270 	/**
271 	 * Gets the offer.
272 	 *
273 	 * @param offerId the offer id
274 	 * @return the offer
275 	 */
276 	public static OfferForm getOffer(final long offerId) {
277 		OfferForm result = null;
278 		try {
279 			final AHOffer offer = AHOfferLocalServiceUtil.getAHOffer(offerId);
280 			result = getOffer(offer);
281 		} catch (final PortalException e) {
282 			m_objLog.error(e);
283 		} catch (final SystemException e) {
284 			m_objLog.error(e);
285 		}
286 		return result;
287 	}

289 	/**
290 	 * Gets the offer for edit.
291 	 *
292 	 * @param offerId the offer id
293 	 * @return the offer for edit
294 	 */
295 	public static OfferForm getOfferForEdit(final long offerId) {
296 		final OfferForm result = getOffer(offerId);
297 
298 		if (result != null) {
299 			final String countryName = result.getRegionCountry();
300 
301 			AHCategories cat = null;
302 			try {
303 				cat = AHCategoriesLocalServiceUtil.getCategory(countryName,
304 				        E_CategoryType.COUNTRIES);
305 				if (cat != null) {
306 					result.setRegionCountry(Long.toString(cat.getCatId()));
307 				}
308 			} catch (final Throwable t) {
309 			}
310 			final String offerTimeName = result.getWorkHours();
311 			try {
312 				cat = AHCategoriesLocalServiceUtil.getCategory(offerTimeName,
313 				        E_CategoryType.OFFERTIME);
314 				if (cat != null) {
315 					result.setWorkHours(Long.toString(cat.getCatId()));
316 				}
317 			} catch (final Throwable t) {
318 			}
319 		}
320 
321 		return result;
322 	}
