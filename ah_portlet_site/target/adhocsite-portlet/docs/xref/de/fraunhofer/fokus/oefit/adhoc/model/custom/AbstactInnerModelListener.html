22  import com.liferay.portal.ModelListenerException;
23  import com.liferay.portal.kernel.log.Log;
24  import com.liferay.portal.kernel.log.LogFactoryUtil;
25  import com.liferay.portal.model.BaseModel;
26  import com.liferay.portal.model.BaseModelListener;
27  
28  // TODO: Auto-generated Javadoc
29  /**
30   * The listener interface for receiving abstactInnerModel events.
31   * The class that is interested in processing a abstactInnerModel
32   * event implements this interface, and the object created
33   * with that class is registered with a component using the
34   * component's <code>addAbstactInnerModelListener<code> method. When
35   * the abstactInnerModel event occurs, that object's appropriate
36   * method is invoked.
37   *
38   * @param <T> the generic type
39   * @see AbstactInnerModelEvent
40   */
41  public abstract class AbstactInnerModelListener<T extends BaseModel<T>> extends
42          BaseModelListener<T> {
43  
44  	private static final Log	              m_objLog	  = LogFactoryUtil
45  	                                                              .getLog(AbstactInnerModelListener.class);
46  	private final ModelNotificationRepository	m_objRepo	= ModelNotificationRepository
47  	                                                              .getInstance();
48  
49  	/* (non-Javadoc)
50  	 * @see com.liferay.portal.model.BaseModelListener#onAfterCreate(com.liferay.portal.model.BaseModel)
51  	 */
52  	@Override
53  	public void onAfterCreate(final T model) throws ModelListenerException {
54  		m_objLog.trace("onAfterCreate::start");
55  		this.m_objRepo.notifyCreate(model);
56  		super.onAfterCreate(model);
57  		m_objLog.trace("onAfterCreate::end");
58  	}
59  
60  	/* (non-Javadoc)
61  	 * @see com.liferay.portal.model.BaseModelListener#onAfterRemove(com.liferay.portal.model.BaseModel)
62  	 */
63  	@Override
64  	public void onAfterRemove(final T model) throws ModelListenerException {
65  		m_objLog.trace("onAfterRemove::start");
66  		this.m_objRepo.notifyDelete(model);
67  		super.onAfterUpdate(model);
68  		m_objLog.trace("onAfterRemove::end");
69  	}
70  
71  	/* (non-Javadoc)
72  	 * @see com.liferay.portal.model.BaseModelListener#onAfterUpdate(com.liferay.portal.model.BaseModel)
73  	 */
74  	@Override
75  	public void onAfterUpdate(final T model) throws ModelListenerException {
76  		m_objLog.trace("onAfterUpdate::start");
77  		this.m_objRepo.notifyUpdate(model);
78  		super.onAfterUpdate(model);
79  		m_objLog.trace("onAfterUpdate::end");
80  	}
81  
82  }
