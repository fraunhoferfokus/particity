85  	/**
86  	 * Validates a given form object and changes user settings accordingly
87  	 *
88  	 * @param oldUser the old Liferay user model
89  	 * @param newProfile the profile containing new values
90  	 * @param result the message handler to bind validation errors to
91  	 */
92  	public static void changeUserProfile(User oldUser,
93  	        final ProfileForm newProfile, final BindingResult result) {
94  		final String oldMail = oldUser.getEmailAddress();
95  		final String newMail = newProfile.getMail();
96  		if (!oldMail.toLowerCase().trim().equals(newMail.toLowerCase().trim())) {
97  			m_objLog.debug("User mail changed!");
98  			try {
99  				oldUser = UserLocalServiceUtil.updateEmailAddress(
100 				        oldUser.getUserId(), oldUser.getPassword(),
101 				        newMail.trim(), newMail.trim());
102 			} catch (final Throwable e) {
103 				result.rejectValue("mail", "notExistentErrorCode",
104 				        "common.form.profile.field.mail.unknown");
105 				e.printStackTrace();
106 			}
107 			if (oldUser != null) {
108 				AHOrgLocalServiceUtil.updateOwner(oldMail, newMail.trim());
109 			}
110 		} else {
111 			m_objLog.debug("User mail not changed!");
112 		}
113 		if (newProfile.getPass1().equals(newProfile.getPass2())) {
114 			if (newProfile.getPass1().replaceAll("\\*", "").length() > 0) {
115 				m_objLog.debug("User password changed!");
116 				try {
117 					UserLocalServiceUtil.updatePassword(oldUser.getUserId(),
118 					        newProfile.getPass1(),
119 					        newProfile.getPass1(), false);
120 				} catch (final Throwable e) {
121 					result.rejectValue("mail", "notExistentErrorCode",
122 					        "common.form.profile.field.pass1.unknown");
123 					m_objLog.error(e);
124 				}
125 
126 			} else {
127 				m_objLog.debug("User password not changed!");
128 			}
129 		}
130 	}

132 	/**
133 	 * Provides an entry point to initialize basic settings and data-structures.
134 	 * Due to use of Liferay's document library and permission settings, this
135 	 * should be run only on login by an administrator. 
136 	 *
137 	 * @param themeDisplay the theme display of the administrator
138 	 */
139 	public static synchronized void checkInit(final ThemeDisplay themeDisplay) {
140 		if (!isSystemInitialized()) {
141 
142 			try {
143 
144 				final long groupId = themeDisplay.getScopeGroupId();
145 				final long userId = themeDisplay.getUserId();
146 
147 				final ServiceContext ctx = new ServiceContext();
148 				ctx.setUserId(userId);
149 				ctx.setScopeGroupId(groupId);
150 				ctx.setAddGroupPermissions(true);
151 				ctx.setAddGuestPermissions(true);
152 				// ctx.setWorkflowAction(WorkflowConstants.ACTION_PUBLISH);
153 
154 				Folder parent = null;
155 				try {
156 					parent = DLAppServiceUtil.getFolder(groupId, 0,
157 					        Constants.ORGANISATION_LOGO_FOLDER);
158 				} catch (final Throwable t) {
159 				}
160 
161 				if (parent == null) {
162 					parent = DLAppServiceUtil.addFolder(groupId, 0,
163 					        Constants.ORGANISATION_LOGO_FOLDER, "", ctx);
164 					m_objLog.info("Folder "
165 					        + Constants.ORGANISATION_LOGO_FOLDER
166 					        + " created ....");
167 				} else {
168 					m_objLog.info("Folder "
169 					        + Constants.ORGANISATION_LOGO_FOLDER
170 					        + " already existing ....");
171 				}
172 
173 				final String[] actionids = new String[2];
174 				final Role guestRole = RoleLocalServiceUtil.getRole(
175 				        PortalUtil.getDefaultCompanyId(), RoleConstants.GUEST);
176 				actionids[0] = ActionKeys.VIEW;
177 				actionids[1] = ActionKeys.ACCESS;
178 				// actionids[2] = ActionKeys.ADD_FILE;
179 
180 				ResourcePermissionServiceUtil.setIndividualResourcePermissions(
181 				        parent.getGroupId(), parent.getCompanyId(),
182 				        DLFolder.class.getName(), parent.getFolderId() + "",
183 				        guestRole.getRoleId(), actionids);
184 
185 				setConfig(E_ConfigKey.INITFLAG, "true");
186 				m_bInitialized = true;
187 
188 			} catch (final Throwable t) {
189 				m_objLog.error(t);
190 			}
191 
192 		}
193 	}

195 	/**
196 	 * Check for a specific role name and create a regular role if not existent
197 	 *
198 	 * @param companyId the company Id of the given role
199 	 * @param roleName the role name
200 	 * @return the role
201 	 */
202 	public static Role checkRole(final long companyId, final String roleName) {
203 		Role result = null;
204 
205 		Role role = null;
206 		try {
207 			role = RoleLocalServiceUtil.fetchRole(companyId,
208 			        Constants.DEFAULT_ROLE_ORGANIZATION);
209 			result = role;
210 		} catch (final SystemException e) {
211 		}
212 		if (role == null) {
213 			try {
214 				role = RoleLocalServiceUtil.createRole(CounterLocalServiceUtil
215 				        .increment(Role.class.getName()));
216 				role.setName(roleName);
217 				role.setCompanyId(companyId);
218 				role.setDescription(roleName);
219 				role.setTitle(roleName);
220 				role.setType(RoleConstants.TYPE_REGULAR);
221 				result = RoleLocalServiceUtil.updateRole(role);
222 			} catch (final Throwable t) {
223 				m_objLog.error(t);
224 			}
225 		}
226 
227 		return result;
228 	}

230 	/**
231 	 * Creates a new portal user
232 	 *
233 	 * @param firstName the first name
234 	 * @param lastName the last name
235 	 * @param mail the email address
236 	 * @param companyId the company id
237 	 * @param locale the user's locale
238 	 * @return the user object
239 	 */
240 	public static User createPortalUser(final String firstName,
241 	        final String lastName, final String mail, final long companyId,
242 	        final Locale locale) {
243 
244 		User user = null;
245 		try {
246 			final Role orgRole = checkRole(companyId,
247 			        Constants.DEFAULT_ROLE_ORGANIZATION);
248 
249 			try {
250 				user = UserLocalServiceUtil.getUserByEmailAddress(companyId,
251 				        mail);
252 			} catch (final Throwable t) {
253 			}
254 
255 			if (user == null) {
256 				user = UserLocalServiceUtil.addUser(0, companyId, true, null,
257 				        null,
258 				        true, null, mail, 0L, "", locale, firstName, "",
259 				        lastName,
260 				        0, 0, false, 0, 1, 1970, "", null, null,
261 				        new long[] { orgRole.getRoleId() }, null, true,
262 				        new ServiceContext());
263 			}
264 
265 		} catch (final Exception e) {
266 			m_objLog.error(e);
267 		}
268 		return user;
269 	}

271 	/**
272 	 * Gets the value for a given configuration key (or the default setting if not found)
273 	 * This method may either access the portlet data-source for local settings or the portal properties for global configuration settings
274 	 *
275 	 * @param key the configuration key
276 	 * @return the configuration value
277 	 */
278 	public static String getConfigValue(final E_ConfigKey key) {
279 		String value = null;
280 		if (key.isSystemProperty()) {
281 			try {
282 				value = PrefsPropsUtil.getString(
283 				        PortalUtil.getDefaultCompanyId(),
284 				        key.getSystemProperty());
285 				// m_objLog.debug("Got portal property "+key.getSystemProperty()+" = "+value);
286 			} catch (final SystemException e) {
287 				value = key.getDefaultValue();
288 				m_objLog.error(e);
289 			}
290 		} else {
291 			value = AHConfigLocalServiceUtil.getConfig(key.toString(),
292 			        key.getDefaultValue());
293 			// m_objLog.debug("Got portlet property "+key.toString()+" = "+value);
294 		}
295 		return value;
296 	}

299 	 * Check whether a given user exists
300 	 *
301 	 * @param companyId the company id
302 	 * @param mail the email address of the user in question
303 	 * @return true, if the user exists, false otherwise
304 	 */
305 	public static boolean hasUser(final long companyId, final String mail) {
306 		boolean result = false;
307 
308 		try {
309 			final User user = UserLocalServiceUtil.getUserByEmailAddress(
310 			        companyId,
311 			        mail);
312 			result = user != null;
313 		} catch (final NoSuchUserException e) {
314 			// IGNORE
315 		} catch (final PortalException e) {
316 			m_objLog.error(e);
317 		} catch (final SystemException e) {
318 			m_objLog.error(e);
319 		}
320 
321 		return result;
322 	}

325 	 * Checks if is system initialized. Uses a portlet configuration setting to determine, if {@link #checkInit()) checkInit} was called before.
326 	 *
327 	 * @return true, if {@link #checkInit()) checkInit} was successfully called before
328 	 */
329 	public static boolean isSystemInitialized() {
330 		if (m_bInitialized == null) {
331 			final String cfgVal = getConfigValue(E_ConfigKey.INITFLAG);
332 			m_bInitialized = cfgVal != null && cfgVal.equals("true");
333 		}
334 		return m_bInitialized;
335 	}

337 	/**
338 	 * Save a set of configuration changes right from the portlet request
339 	 * 
340 	 * The request is parsed for configuration values with names out of <code>E_ConfigKey</code>
341 	 * 
342 	 * As a result, social media plugins and messagec composers will be reinitialized to address the changes
343 	 *
344 	 * @param request the portlet request
345 	 */
346 	public static void saveConfig(final ActionRequest request) {
347 		final Enumeration<String> paramNames = request.getParameterNames();
348 		while (paramNames.hasMoreElements()) {
349 			final String keyName = paramNames.nextElement();
350 			E_ConfigKey key = null;
351 			try {
352 				key = E_ConfigKey.valueOf(keyName);
353 			} catch (final Throwable t) {
354 			}
355 			if (key != null) {
356 
357 				final String value = request.getParameter(keyName);
358 				if (value != null) {
359 					// m_objLog.debug("Accepting save parameter: "+keyName+" with value: "+value);
360 					final String oldVal = getConfigValue(key);
361 					if (!oldVal.equals(value)) {
362 						m_objLog.debug("Setting changed parameter: " + keyName);
363 						setConfig(key, value);
364 					}
365 
366 				} /*
367 				 * else { m_objLog.debug("Accepting save parameter: "+keyName+
368 				 * " with empty value"); }
369 				 */
370 			} /*
371 			 * else { m_objLog.debug("Ignoring save parameter: "+keyName); }
372 			 */
373 		}
374 		/*
375 		 * refresh modules that depend on fresh configuration settings
376 		 */
377 		MessageComposer.getInstance().refreshCfg();
378 		SocialMediaFactory.refreshCfg();
379 	}

384 	 * @param key the configuration key
385 	 * @param value the configuration value

387 	public static void setConfig(final E_ConfigKey key, final String value) {
388 		if (key.isSystemProperty()) {
389 			updatePortalPreferences(key.getSystemProperty(), value);
390 		} else {
391 			AHConfigLocalServiceUtil.setConfig(key.toString(), value);
392 		}

396 	/**
397 	 * Update portal preferences.
398 	 *
399 	 * @param key the property name
400 	 * @param value the value
401 	 */
402 	private static void updatePortalPreferences(final String key,
403 	        final String value) {
404 		try {
405 			final List<PortalPreferences> prefs = PortalPreferencesLocalServiceUtil
406 			        .getPortalPreferenceses(QueryUtil.ALL_POS,
407 			                QueryUtil.ALL_POS);
408 
409 			for (final PortalPreferences pref : prefs) {
410 				final long ownerId = pref.getOwnerId();
411 				final int ownerType = pref.getOwnerType();
412 				// m_objLog.debug("Pref is "+ownerId+","+ownerType+","+pref.getPortalPreferencesId());
413 				final PortletPreferences pps = PortalPreferencesLocalServiceUtil
414 				        .getPreferences(ownerId, ownerType);
415 				final Enumeration<String> names = pps.getNames();
416 				while (names.hasMoreElements()) {
417 					final String pkey = names.nextElement();
418 					if (pkey.equals(key)) {
419 						// m_objLog.debug("Found property "+key+" with former value "+pps.getValue(key,"N/A"));
420 						pps.setValue(pkey, value);
421 						pps.store();
422 						break;
423 					}
424 				}
425 			}
426 		} catch (final Throwable e) {
427 			m_objLog.error(e);
428 		}
